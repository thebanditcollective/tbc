<script>
  (function() {
    const track = document.getElementById('bcLookbookTrack');
    const cards = track.querySelectorAll('.bc-lookbook-card');
    const dotsContainer = document.getElementById('bcLookbookDots');
    const prevBtn = document.querySelector('.bc-lookbook-prev');
    const nextBtn = document.querySelector('.bc-lookbook-next');
    let hasSwiped = false;

    // Create dots
    cards.forEach((_, i) => {
      const dot = document.createElement('div');
      dot.className = 'bc-lookbook-dot';
      if (i === 0) dot.classList.add('active');
      dotsContainer.appendChild(dot);
    });
    const dots = dotsContainer.querySelectorAll('.bc-lookbook-dot');

    function updateActive() {
      let activeIndex = 0;
      let minDiff = Infinity;

      cards.forEach((card, i) => {
        const rect = card.getBoundingClientRect();
        const trackRect = track.getBoundingClientRect();
        const diff = Math.abs(rect.left - trackRect.left);
        if (diff < minDiff) {
          minDiff = diff;
          activeIndex = i;
        }
      });

      cards.forEach((c, i) => c.classList.toggle('is-active', i === activeIndex));
      dots.forEach((d, i) => d.classList.toggle('active', i === activeIndex));

      // Hide mobile indicator after first swipe
      if (!hasSwiped && activeIndex > 0) {
        hasSwiped = true;
        document.querySelectorAll('.bc-lookbook-indicator svg').forEach(svg => {
          svg.style.animation = 'none';
        });
        document.querySelectorAll('.bc-lookbook-indicator').forEach(ind => {
          ind.style.transition = 'opacity 0.4s ease';
          ind.style.opacity = '0';
        });
      }

      // Desktop arrow visibility
      if (prevBtn && nextBtn) {
        const isAtStart = track.scrollLeft < 10;
        const isAtEnd = track.scrollLeft + track.clientWidth >= track.scrollWidth - 10;

        prevBtn.classList.toggle('is-hidden', isAtStart);
        nextBtn.classList.toggle('is-hidden', isAtEnd);
      }
    }

    // Scroll one card width on arrow click
    function scrollByCard(direction) {
      const cardWidth = cards[0].offsetWidth + 12; // card + gap
      track.scrollBy({ left: direction * cardWidth, behavior: 'smooth' });
    }

    if (prevBtn) {
      prevBtn.addEventListener('click', () => scrollByCard(-1));
    }

    if (nextBtn) {
      nextBtn.addEventListener('click', () => scrollByCard(1));
    }

    track.addEventListener('scroll', () => {
      window.requestAnimationFrame(updateActive);
    });

    // Init
    updateActive();
  })();
</script>
